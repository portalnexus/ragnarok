Python - Rich

from rich.console import Console
from rich.live import Live
from rich.spinner import Spinner
from rich.text import Text
import time

console = Console()

# Simple spinner
spinner = Spinner("dots", text="Loading...")
console.print(spinner)

# Animated spinner with Live
with Live(Spinner("dots", text="Processing..."), 
          refresh_per_second=10) as live:
    time.sleep(3)
    live.update(Spinner("dots", text="Almost done...", 
                        style="green"))
    time.sleep(1)

Python - Custom Spinner

import sys
import time
import itertools

def spinner(message="Loading", duration=0.1):
    frames = ["⠋", "⠙", "⠹", "⠸", "⠼", "⠴", "⠦", "⠧", "⠇", "⠏"]
    spinner_iter = itertools.cycle(frames)
    
    try:
        while True:
            frame = next(spinner_iter)
            sys.stdout.write(f"\r{frame} {message}")
            sys.stdout.flush()
            time.sleep(duration)
            yield
    except KeyboardInterrupt:
        sys.stdout.write("\r✓ Done!\n")
        sys.stdout.flush()

# Usage
for _ in spinner("Processing data"):
    # Your work here
    if done:
        break

Rust - Ratatui Spinner

use ratatui::{
    Frame,
    widgets::{Block, Borders, Paragraph},
};

fn render_spinner(frame: &mut Frame, area: Rect, frame_num: u64) {
    let spinner = ["⠋", "⠙", "⠹", "⠸", "⠼", "⠴", "⠦", "⠧", "⠇", "⠏"];
    let current = spinner[frame_num as usize % spinner.len()];
    
    let paragraph = Paragraph::new(format!("{} Loading...", current))
        .block(Block::default().borders(Borders::ALL));
    
    frame.render_widget(paragraph, area);
}

Go - Bubble Tea Spinner

import (
    "github.com/charmbracelet/bubbles/spinner"
    tea "github.com/charmbracelet/bubbletea"
)

type model struct {
    spinner spinner.Model
}

func (m model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
    var cmd tea.Cmd
    m.spinner, cmd = m.spinner.Update(msg)
    return m, cmd
}

func (m model) View() string {
    return m.spinner.View() + " Loading..."
}

// Spinner types: Line, Dot, MiniDot, Jump, 
// Pulse, Points, Globe, Moon, Monkey

Bash - Simple Spinner

#!/bin/bash

spinner() {
    local pid=$1
    local delay=0.1
    local spinstr='|/-\'
    
    while [ "$(ps a | awk '{print $1}' | grep -w $pid)" ]; do
        local temp=${spinstr#?}
        printf " [%c] Loading..." "$spinstr"
        local spinstr=$temp${spinstr%"$temp"}
        sleep $delay
        printf "\b\b\b\b\b\b\b\b\b\b\b\b\b\b\b\b\b\b"
    done
    printf "    \r"
}

# Usage
long_command &
spinner $!
echo "Done!"
