Progress Bars

Progress bar implementations for long-running tasks

Python - Rich Progress

from rich.progress import (
    Progress, 
    BarColumn, 
    TextColumn, 
    TimeRemainingColumn,
    DownloadColumn,
    TransferSpeedColumn
)
import time

# Basic progress bar
with Progress() as progress:
    task = progress.add_task("Processing...", total=100)
    while not progress.finished:
        progress.update(task, advance=1)
        time.sleep(0.02)

# Advanced with multiple columns
with Progress(
    TextColumn("[bold blue]{task.description}"),
    BarColumn(bar_width=40),
    TextColumn("[progress.percentage]{task.percentage:>3.1f}%"),
    TextColumn("•"),
    TimeRemainingColumn(),
    TextColumn("•"),
    DownloadColumn(),
) as progress:
    task = progress.add_task("Downloading", total=1000000)
    # Your download code here

Python - tqdm

from tqdm import tqdm
import time

# Basic usage
for i in tqdm(range(100)):
    time.sleep(0.01)

# With description
for i in tqdm(range(100), desc="Processing"):
    time.sleep(0.01)

# Manual update
with tqdm(total=100) as pbar:
    for i in range(10):
        time.sleep(0.1)
        pbar.update(10)

# Nested progress bars
with tqdm(total=100, desc="Outer") as pbar1:
    for i in range(10):
        with tqdm(total=10, desc="Inner", leave=False) as pbar2:
            for j in range(10):
                time.sleep(0.01)
                pbar2.update(1)
        pbar1.update(1)

Node.js - CLI Progress

import cliProgress from 'cli-progress';

// Basic progress bar
const bar = new cliProgress.SingleBar({}, 
    cliProgress.Presets.shades_classic);

bar.start(100, 0);
for (let i = 0; i < 100; i++) {
    bar.update(i);
    await sleep(10);
}
bar.stop();

// With options
const bar = new cliProgress.SingleBar({
    format: '{bar} {percentage}% | {value}/{total}',
    barCompleteChar: '█',
    barIncompleteChar: '░',
    hideCursor: true
});

// Multi progress
const multibar = new cliProgress.MultiBar({
    format: '{bar} {percentage}% | {label}'
});

const bar1 = multibar.create(100, 0, {label: 'Download'});
const bar2 = multibar.create(100, 0, {label: 'Extract'});

Rust - Indicatif

use indicatif::{ProgressBar, ProgressStyle};
use std::thread;
use std::time::Duration;

// Basic progress bar
let pb = ProgressBar::new(100);
for i in 0..100 {
    thread::sleep(Duration::from_millis(10));
    pb.set_position(i);
}
pb.finish();

// With style
let pb = ProgressBar::new(100);
pb.set_style(ProgressStyle::default_bar()
    .template("{spinner:.green} [{elapsed_precise}] [{bar:40.cyan/blue}] {pos}/{len} ({eta})")
    .unwrap()
    .progress_chars("#>-"));

// Multiple bars
let m = indicatif::MultiProgress::new();
let pb1 = m.add(ProgressBar::new(100));
let pb2 = m.add(ProgressBar::new(100));

Go - Progress Bar

import (
    "github.com/schollz/progressbar/v3"
    "time"
)

// Basic progress bar
bar := progressbar.Default(100)
for i := 0; i < 100; i++ {
    bar.Add(1)
    time.Sleep(10 * time.Millisecond)
}

// With options
bar := progressbar.NewOptions(1000,
    progressbar.OptionSetDescription("Downloading"),
    progressbar.OptionSetWidth(40),
    progressbar.OptionShowCount(),
    progressbar.OptionShowIts(),
    progressbar.OptionSetItsString("items"),
    progressbar.OptionSetPredictTime(true),
    progressbar.OptionSetTheme(progressbar.Theme{
        BarStart: "[",
        BarEnd: "]",
        Current: "█",
        CurrentEmpty: "░",
    }),
)

// Spinner for unknown total
spinner := progressbar.NewSpinner("Processing...")
for {
    spinner.Add(1)
    time.Sleep(100 * time.Millisecond)
}

Bash - Progress Bar

#!/bin/bash

progress_bar() {
    local total=$1
    local current=0
    
    while read -r line; do
        ((current++))
        local percent=$((current * 100 / total))
        local filled=$((percent / 5))
        local empty=$((20 - filled))
        
        printf "\r[%${filled}s%${empty}s] %d%%" \
            | tr ' ' '#' | tr ' ' '-' \
            "$current" "$percent"
    done
}

# Usage
for i in {1..100}; do
    echo "$i"
    sleep 0.05
done | progress_bar 100
echo

Bar Characters

Common characters for progress bars:

# Block characters

# Line characters
─ │ ━ ┃ ─━ ┃━

# Arrows
← ↑ → ↓

Classic:  [████████░░░░]
