Go Bubble Tea

The Elm Architecture for terminal UIs

Overview

Bubble Tea is a functional Go framework for building terminal applications based on The Elm Architecture. Created by Charmbracelet, it provides a clean, composable way to build TUIs with proper state management.

Installation

go mod init myapp
go get github.com/charmbracelet/bubbletea

The Elm Architecture

Bubble Tea follows three core concepts:

  • Model - Your application state
  • Update - How state changes based on messages
  • View - How state is rendered

Hello World

package main

import (
    "fmt"
    "os"
    "github.com/charmbracelet/bubbletea"
)

type model struct {
    cursor int
    choice string
}

func (m model) Init() tea.Cmd {
    return nil
}

func (m model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
    switch msg := msg.(type) {
    case tea.KeyMsg:
        switch msg.String() {
        case "ctrl+c", "q":
            return m, tea.Quit
        }
    }
    return m, nil
}

func (m model) View() string {
    return fmt.Sprintf("\n  Hello, %s!\n\n  Press q to quit.\n", m.choice)
}

func main() {
    p := tea.NewProgram(model{choice: "World"})
    if _, err := p.Run(); err != nil {
        fmt.Printf("Error: %v", err)
        os.Exit(1)
    }
}

Messages and Commands

// Messages trigger updates
type tickMsg time.Time
type errMsg error

// Commands are async operations
func tickEvery(d time.Duration) tea.Cmd {
    return tea.Tick(d, func(t time.Time) tea.Msg {
        return tickMsg(t)
    })
}

// In Update, return a command
func (m model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
    switch msg.(type) {
    case tickMsg:
        return m, tickEvery(time.Second)
    }
    return m, nil
}

var (
    titleStyle = lipgloss.NewStyle().
        Bold(true).
        Foreground(lipgloss.Color("#FAFAFA")).
        Background(lipgloss.Color("#7D56F4"))
    
    selectedStyle = lipgloss.NewStyle().
        Foreground(lipgloss.Color("#FFF7DB"))
)

func (m model) View() string {
    return titleStyle.Render("My App")
}
